56 public String generateJwtToken(String userId, String username, boolean rememberMe) {
57 Long expirationTime = rememberMe ? rememberMeExpirationMs : jwtExpirationMs;
58
59 return JWT.create()
60 .withSubject(userId)
61 .withClaim("username", username)
62 .withClaim("rememberMe", rememberMe)
63 .withIssuedAt(new Date())
64 .withExpiresAt(new Date(System.currentTimeMillis() + expirationTime))
65 .sign(Algorithm.HMAC256(jwtSecret));
66 }

127 public String extractTokenFromRequest(HttpServletRequest request) {
128 // First try to get from Authorization header
129 String bearerToken = request.getHeader("Authorization");
130 if (bearerToken != null && bearerToken.startsWith("Bearer ")) {
131 return bearerToken.substring(7);
132 }
133
134 // Then try to get from cookie
135 Cookie[] cookies = request.getCookies();
136 if (cookies != null) {
137 for (Cookie cookie : cookies) {
138 if ("auth_token".equals(cookie.getName())) {
139 return cookie.getValue();
140 }
141 }
