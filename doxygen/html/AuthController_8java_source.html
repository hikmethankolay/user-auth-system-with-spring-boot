76 @PostMapping("/login")

78 // Get client IP address for rate limiting

81 try {
82 // Authentication is performed in the service layer
83 String token = userService.authenticateUser(loginRequest, clientIp);
84
85 if (token != null) {
86 // Login successful

89

91 if (loginRequest.rememberMe()) {

94 .secure(false) // Enable in production with HTTPS
95 .path("/")
96 .maxAge(Duration.ofDays(30))
97 .sameSite("Strict")
98 .build();
99

105 return ResponseEntity.status(HttpStatus.BAD_REQUEST)

120 private String getClientIp(HttpServletRequest request) {
121 String xfHeader = request.getHeader("X-Forwarded-For");
122 if (xfHeader == null || xfHeader.isEmpty() || "unknown".equalsIgnoreCase(xfHeader)) {
123 return request.getRemoteAddr();
124 }
125 return xfHeader.split(",")[0];
