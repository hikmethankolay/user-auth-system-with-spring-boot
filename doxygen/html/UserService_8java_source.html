156 public String authenticateUser(LoginRequestDTO loginRequest, String clientIp) {
157 String identifier = loginRequest.identifier();
158
159 // Check if the IP address is blocked due to too many failed login attempts
160 if (loginAttemptService.isBlocked(clientIp)) {
161 throw new RuntimeException("Too many failed login attempts from this IP. Please try again later.");
162 }
163
164 // Check if the user is blocked due to too many failed attempts
165 if (loginAttemptService.isBlocked(identifier)) {
166 throw new RuntimeException("Account is temporarily locked due to too many failed login attempts. Please try again later.");
167 }
168
169 Optional<User> user = userRepository.findByUsernameOrEmail(identifier, identifier);
170
171 if (user.isPresent() && passwordEncoder.matches(loginRequest.password(), user.get().getPassword())) {
172 // Authentication successful - reset failed attempts counter
173 loginAttemptService.loginSucceeded(clientIp);
174 loginAttemptService.loginSucceeded(identifier);
175
176 return jwtUtils.generateJwtToken(
177 String.valueOf(user.get().getId()),
178 user.get().getUsername(),
179 loginRequest.rememberMe()
180 );
181 } else {
182 // Authentication failed - increment failed attempts counter
183 loginAttemptService.loginFailed(clientIp);
184 loginAttemptService.loginFailed(identifier);
185 return null;
186 }

194 public String refreshToken(String token) {

196 try {
197 Long userId = jwtUtils.getUserIdFromJwtToken(token);
198 Optional<User> userOpt = findById(userId);

200 if (userOpt.isPresent()) {

206 user.getUsername(),
207 wasRememberMe

277 private void checkUserUniqueness(UserDTO userDTO, Long userId) {
