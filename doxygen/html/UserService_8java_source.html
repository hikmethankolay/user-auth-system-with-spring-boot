153 public String authenticateUser(LoginRequestDTO loginRequest, String clientIp) {
154 String identifier = loginRequest.identifier();
155
156 // Check if the IP address is blocked due to too many failed login attempts
157 if (loginAttemptService.isBlocked(clientIp)) {
158 throw new RuntimeException("Too many failed login attempts from this IP. Please try again later.");
159 }
160
161 // Check if the user is blocked due to too many failed attempts
162 if (loginAttemptService.isBlocked(identifier)) {
163 throw new RuntimeException("Account is temporarily locked due to too many failed login attempts. Please try again later.");
164 }
165
166 Optional<User> user = userRepository.findByUsernameOrEmail(identifier, identifier);
167
168 if (user.isPresent() && passwordEncoder.matches(loginRequest.password(), user.get().getPassword())) {
169 // Authentication successful - reset failed attempts counter
170 loginAttemptService.loginSucceeded(clientIp);
171 loginAttemptService.loginSucceeded(identifier);
172
173 return jwtUtils.generateJwtToken(
174 String.valueOf(user.get().getId()),
175 user.get().getUsername(),
176 loginRequest.rememberMe()
177 );
178 } else {
179 // Authentication failed - increment failed attempts counter
180 loginAttemptService.loginFailed(clientIp);
181 loginAttemptService.loginFailed(identifier);
182 return null;
183 }

270 private <T extends UserInfo> void checkUserValidation(T userInfo, Long userId) {
271 Set<ConstraintViolation<T>> violations = validator.validate(userInfo);
272 if (!violations.isEmpty()) {
273 throw new ConstraintViolationException(violations);
274 }
275
276 userRepository.findByUsername(userInfo.getUsername())
277 .filter(user -> !Objects.equals(user.getId(), userId))
278 .ifPresent(user -> { throw new RuntimeException("Username is already taken!"); });
279
280 userRepository.findByEmail(userInfo.getEmail())
281 .filter(user -> !Objects.equals(user.getId(), userId))
282 .ifPresent(user -> { throw new RuntimeException("Email is already taken!"); });
283 }
